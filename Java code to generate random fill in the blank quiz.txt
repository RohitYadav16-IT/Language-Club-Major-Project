import java.util.*;

public class FillInTheBlankQuiz {
    public static void main(String[] args) {
        // Define the quiz questions and their corresponding answers
        String[] questions = {
            "The capital of France is _________.",
            "The world's largest ocean is the _________.",
            "The tallest mountain in the world is _________."
        };
        String[] answers = {"Paris", "Pacific", "Mount Everest"};

        // Create a list to store the indices of the questions
        ArrayList<Integer> indices = new ArrayList<>();
        for (int i = 0; i < questions.length; i++) {
            indices.add(i);
        }

        // Shuffle the indices to generate a random order for the questions
        Collections.shuffle(indices);

        // Print the questions and get the user's answers
        Scanner scanner = new Scanner(System.in);
        for (int i : indices) {
            System.out.println(questions[i]);
            String userAnswer = scanner.nextLine().trim();
            if (userAnswer.equalsIgnoreCase(answers[i])) {
                System.out.println("Correct!");
            } else {
                System.out.println("Incorrect. The correct answer is " + answers[i] + ".");
            }
        }

        // End the quiz
        System.out.println("Quiz complete.");
    }
}
